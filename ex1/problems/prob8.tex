\begin{itemize}
\item{$P[|X-\E[x]>\alpha] \leq \frac{\Var[X]}{\alpha^2}$:
  \begin{align*}
    \frac{\Var[X]}{\alpha}
     &= \frac{\E[(X-\E[x])^2]}{\alpha^2} \\
     &\ge P((X-\E[x])^2>\alpha^2) \\
     &= P(|X-\E[x]|>\alpha)
  \end{align*}
}
\item{$P[|X|>\frac{3}{4}] $:
	\begin{align*}
	P[|X|>\frac{3}{4}] \le \frac{n*\frac{1}{4}}{n*n*\frac{9}{12}} \\
	P[X>\frac{3}{4}] \le \frac{\frac{n*\frac{1}{4}}{n*n*\frac{9}{12}}}{2}
	\end{align*}
}
\end{itemize}
